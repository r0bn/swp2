handles the event if a user choose a story to edit

    $scope.storySelected = false
    $scope.handleStorySelected = (story) ->
        $scope.storySelected = true

        $http.get("http://api.dev.la/story/#{story.id}")
            .success (data) ->
                $scope.xmlFile = data
